/*** BEGIN: Car::allInstances()->collect(c|1 = 1) ***/
SELECT TEMP_body.val AS val, TEMP_body.res AS res, TEMP_body.type AS type
FROM (
    /*** BEGIN: Car::allInstances() ***/
    SELECT 1 AS val, Car_id AS res, 'Car' AS type
    FROM Car
    /*** END: Car::allInstances() ***/
) AS TEMP_src
JOIN (
    /*** BEGIN: 1 = 1 ***/
    SELECT 1 AS val, 'Boolean' AS type, TEMP_LEFT.res = TEMP_RIGHT.res AS res
    FROM (
        /*** BEGIN: 1 ***/
        SELECT 1 AS val, 1 AS res, 'Integer' AS type
        /*** END: 1 ***/
    ) AS TEMP_LEFT,
    (
        /*** BEGIN: 1 ***/
        SELECT 1 AS val, 1 AS res, 'Integer' AS type
        /*** END: 1 ***/
    ) AS TEMP_RIGHT
    /*** END: 1 = 1 ***/
) AS TEMP_body
/*** END: Car::allInstances()->collect(c|1 = 1) ***/
;

/*** BEGIN: Car.allInstances()->collect(c| 1 = 1) ***/
SELECT TEMP_body.val AS val, TEMP_body.res AS res, TEMP_body.type AS type
FROM (
    /*** BEGIN: Car.allInstances() ***/
    SELECT 1 AS val, Car_id AS res, 'Car' AS type
    FROM Car
    /*** END: Car.allInstances() ***/
) AS TEMP_source
JOIN (
    /*** BEGIN: 1 = 1 ***/
    SELECT TEMP_LEFT.res = TEMP_RIGHT.res AS res, 'boolean' AS type, 1 AS val
    FROM (
        /*** BEGIN: 1 ***/
        SELECT 1 AS val, 1 AS res, 'Integer' AS type
        /*** END: 1 ***/
    ) AS TEMP_LEFT
    JOIN (
        /*** BEGIN: 1 ***/
        SELECT 1 AS val, 1 AS res, 'Integer' AS type
        /*** END: 1 ***/
    ) AS TEMP_RIGHT
    /*** END: 1 = 1 ***/
) AS TEMP_body
/*** END: Car.allInstances()->collect(c| 1 = 1) ***/
;
